-- Customers
CREATE TABLE customers (
    customer_id        UUID PRIMARY KEY,
    first_name         VARCHAR(100),
    last_name          VARCHAR(100),
    email              VARCHAR(255) UNIQUE NOT NULL,
    phone_number       VARCHAR(50),
    date_of_birth      DATE,
    customer_segment   VARCHAR(50), -- e.g., VIP, Regular, Discount
    lifetime_value     DECIMAL(12,2), -- total revenue generated by customer
    created_at         TIMESTAMP,
    updated_at         TIMESTAMP
);

-- Addresses
CREATE TABLE addresses (
    address_id         UUID PRIMARY KEY,
    customer_id        UUID REFERENCES customers(customer_id),
    street             VARCHAR(255),
    city               VARCHAR(100),
    postal_code        VARCHAR(20),
    country            VARCHAR(100),
    is_default         BOOLEAN,
    created_at         TIMESTAMP
);

-- User Accounts / Authentication
CREATE TABLE user_accounts (
    account_id         UUID PRIMARY KEY,
    customer_id        UUID REFERENCES customers(customer_id),
    username           VARCHAR(100) UNIQUE NOT NULL,
    password_hash      VARCHAR(255) NOT NULL,
    last_login         TIMESTAMP,
    failed_login_count INT,
    is_locked          BOOLEAN
);

-- Products
CREATE TABLE products (
    product_id         UUID PRIMARY KEY,
    product_name       VARCHAR(255),
    description        TEXT,
    category           VARCHAR(100),
    price              DECIMAL(10,2),        -- selling price
    cost_price         DECIMAL(10,2),        -- internal procurement cost
    margin             DECIMAL(5,2),         -- calculated margin %
    supplier_id        UUID,
    stock_quantity     INT,
    created_at         TIMESTAMP
);

-- Suppliers (Business sensitive)
CREATE TABLE suppliers (
    supplier_id        UUID PRIMARY KEY,
    supplier_name      VARCHAR(255),
    contact_email      VARCHAR(255),
    contract_terms     TEXT,                 -- confidential agreements
    rating             DECIMAL(3,2),
    created_at         TIMESTAMP
);

-- Orders
CREATE TABLE orders (
    order_id           UUID PRIMARY KEY,
    customer_id        UUID REFERENCES customers(customer_id),
    order_date         TIMESTAMP,
    order_status       VARCHAR(50),
    total_amount       DECIMAL(12,2),
    discount_amount    DECIMAL(10,2),
    tax_amount         DECIMAL(10,2),
    net_amount         DECIMAL(12,2),        -- revenue after discounts/taxes
    profit_amount      DECIMAL(12,2),        -- calculated profit
    currency           VARCHAR(10),
    sales_channel      VARCHAR(50),          -- e.g., Web, Mobile, Marketplace
    shipping_address_id UUID REFERENCES addresses(address_id),
    billing_address_id  UUID REFERENCES addresses(address_id)
);

-- Order Items
CREATE TABLE order_items (
    order_item_id      UUID PRIMARY KEY,
    order_id           UUID REFERENCES orders(order_id),
    product_id         UUID,
    quantity           INT,
    price_per_unit     DECIMAL(10,2),
    cost_per_unit      DECIMAL(10,2),        -- internal cost reference
    discount           DECIMAL(10,2),
    total_price        DECIMAL(12,2),
    total_cost         DECIMAL(12,2),
    item_profit        DECIMAL(12,2)
);

-- Payments
CREATE TABLE payments (
    payment_id         UUID PRIMARY KEY,
    order_id           UUID REFERENCES orders(order_id),
    payment_method     VARCHAR(50),
    card_holder_name   VARCHAR(255),
    card_number        VARCHAR(50),
    card_expiry_date   VARCHAR(10),
    card_cvv           VARCHAR(10),
    iban               VARCHAR(50),
    payment_status     VARCHAR(50),
    payment_provider   VARCHAR(100),         -- e.g., Stripe, PayPal
    transaction_fee    DECIMAL(10,2),        -- cost per transaction
    payment_date       TIMESTAMP
);

-- Shipment Tracking
CREATE TABLE shipments (
    shipment_id        UUID PRIMARY KEY,
    order_id           UUID REFERENCES orders(order_id),
    tracking_number    VARCHAR(100),
    carrier            VARCHAR(100),
    shipping_cost      DECIMAL(10,2),        -- business sensitive logistics cost
    shipped_date       TIMESTAMP,
    delivery_date      TIMESTAMP,
    status             VARCHAR(50)
);

-- Reviews
CREATE TABLE reviews (
    review_id          UUID PRIMARY KEY,
    product_id         UUID REFERENCES products(product_id),
    customer_id        UUID REFERENCES customers(customer_id),
    rating             INT,
    comment            TEXT,
    sentiment_score    DECIMAL(3,2),         -- derived analytics
    created_at         TIMESTAMP
);

-- Session / Tracking Data
CREATE TABLE user_sessions (
    session_id         UUID PRIMARY KEY,
    customer_id        UUID REFERENCES customers(customer_id),
    ip_address         VARCHAR(50),
    user_agent         TEXT,
    session_start      TIMESTAMP,
    session_end        TIMESTAMP,
    conversion_flag    BOOLEAN               -- did session result in purchase
);

-- Wishlist
CREATE TABLE wishlist (
    wishlist_id        UUID PRIMARY KEY,
    customer_id        UUID REFERENCES customers(customer_id),
    product_id         UUID REFERENCES products(product_id),
    added_at           TIMESTAMP
);

-- Sales Analytics (Aggregated - highly business sensitive)
CREATE TABLE sales_metrics (
    metric_id          UUID PRIMARY KEY,
    date               DATE,
    total_revenue      DECIMAL(15,2),
    total_profit       DECIMAL(15,2),
    total_orders       INT,
    average_order_value DECIMAL(10,2),
    conversion_rate    DECIMAL(5,2),
    customer_acquisition_cost DECIMAL(10,2)
);

-- Product Performance Metrics
CREATE TABLE product_performance (
    performance_id     UUID PRIMARY KEY,
    product_id         UUID REFERENCES products(product_id),
    date               DATE,
    units_sold         INT,
    revenue            DECIMAL(12,2),
    profit             DECIMAL(12,2),
    return_rate        DECIMAL(5,2)
);
